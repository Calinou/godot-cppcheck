	String documentation;

	const void *type_tag;

	bool is_tool;

	inline NativeScriptDesc() :<--- Member variable 'NativeScriptDesc::base_data' is not initialized in the constructor.<--- Member variable 'NativeScriptDesc::is_tool' is not initialized in the constructor.
