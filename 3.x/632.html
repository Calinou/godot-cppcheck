		return bounds_grid[(p_z * bounds_grid_width) + p_x];
	}

		return heights[(p_z * width) + p_x];

struct MotionShapeSW : public ShapeSW {
	ShapeSW *shape;
	Vector3 motion;

	virtual PhysicsServer::ShapeType get_type() const { return PhysicsServer::SHAPE_CONVEX_POLYGON; }<--- Function in derived class

	void project_range(const Vector3 &p_normal, const Transform &p_transform, real_t &r_min, real_t &r_max) const {<--- Function in derived class
		Vector3 cast = p_transform.basis.xform(motion);
		real_t mina, maxa;
		real_t minb, maxb;
		Transform ofsb = p_transform;
		ofsb.origin += cast;
		shape->project_range(p_normal, p_transform, mina, maxa);
		shape->project_range(p_normal, ofsb, minb, maxb);
		r_min = MIN(mina, minb);
		r_max = MAX(maxa, maxb);
	}

	Vector3 get_support(const Vector3 &p_normal) const {<--- Function in derived class
		Vector3 support = shape->get_support(p_normal);
		if (p_normal.dot(motion) > 0) {
			support += motion;
		}
		return support;
	}
	virtual void get_supports(const Vector3 &p_normal, int p_max, Vector3 *r_supports, int &r_amount, FeatureType &r_type) const { r_amount = 0; }<--- Function in derived class
	bool intersect_segment(const Vector3 &p_begin, const Vector3 &p_end, Vector3 &r_result, Vector3 &r_normal) const { return false; }<--- Function in derived class
	virtual bool intersect_point(const Vector3 &p_point) const { return false; }<--- Function in derived class
	virtual Vector3 get_closest_point_to(const Vector3 &p_point) const { return p_point; }<--- Function in derived class

	Vector3 get_moment_of_inertia(real_t p_mass) const { return Vector3(); }<--- Function in derived class

	virtual void set_data(const Variant &p_data) {}<--- Function in derived class
	virtual Variant get_data() const { return Variant(); }<--- Function in derived class

	MotionShapeSW() { configure(AABB()); }<--- Member variable 'MotionShapeSW::shape' is not initialized in the constructor.
};
