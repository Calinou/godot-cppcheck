	};

	virtual int get_input_value_port_count() const;<--- Function in derived class
	virtual int get_output_value_port_count() const;<--- Function in derived class

	virtual PropertyInfo get_input_value_port_info(int p_idx) const;<--- Function in derived class
	virtual PropertyInfo get_output_value_port_info(int p_idx) const;<--- Function in derived class

	virtual String get_caption() const;<--- Function in derived class
	virtual String get_category() const;<--- Function in derived class

	void set_deconstruct_type(Variant::Type p_type);
	Variant::Type get_deconstruct_type() const;

	virtual VisualScriptNodeInstance *instance(VisualScriptInstance *p_instance);<--- Function in derived class

	VisualScriptDeconstruct();
};
